const http = require('http'); // Node.js built-in module for HTTP
const url = require('url');   // Node.js built-in module for URL parsing


const rootContentGenerator = require('./rootModule');
const aboutContentGenerator = require('./aboutModule');
const contactContentGenerator = require('./contactModule');

const PORT = 5000; // You can change this port number if it's already in use

const server = http.createServer((req, res) => {
    const parsedUrl = url.parse(req.url, true); // true to parse query string as object
    const path = parsedUrl.pathname;

    res.writeHead(200, { 'Content-Type': 'text/html' }); // Set content type to HTML


    const userName = "John Smith";

    switch (path) {
        case '/':
            res.write('<h1>Welcome to my Node.js Application</h1>');
            res.write(`<p>${rootContentGenerator.getRootContent(userName)}</p>`);
            break;
        case '/about':
            res.write('<h1>This is the About Page</h1>');
            res.write(`<p>${aboutContentGenerator.getAboutContent(userName)}</p>`);
            break;
        case '/contact':
            res.write('<h1>This is the Contact Page</h1>');
            res.write(`<p>${contactContentGenerator.getContactContent(userName)}</p>`);
            break;
        case '/gallery':
            res.write('<h1>This is the Gallery Page</h1>');
            break;
        default:
            res.writeHead(400, { 'Content-Type': 'text/html' }); // Set status to Bad Request
            res.write('<h1>Invalid Request</h1>');
            break;
    }
    res.end(); // End the response
});

server.listen(PORT, () => {
    console.log(`Server running at http://localhost:${PORT}/`);
});
